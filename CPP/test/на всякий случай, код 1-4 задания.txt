using namespace std;

int main()
{
	setlocale(LC_ALL, "rus");
	int a, b;
	char c;
	cout << "Введите длинну прямоугольника: ";
	cin >> a;
	cout << "Введите ширину прямоугольника: ";
	cin >> b;
	cout << "Введите символ для прямоугальника: ";
	cin >> c;
	for (int i = 1; i <= a; i++) {
		for (int j = 1; j <= b; j++) {
			if ((i == 1) || (j == 1) || (i == a) || (j == b)) {
				cout << c;
			}
			else {
				cout << c;
			}
		}
		cout << endl;
	}
	return 0;
}

#include <iostream>
using namespace std;

int main()
{
	setlocale(LC_ALL, "rus");
	int a, r;
	cout << "Введите длинну ромба: "; // нечётное число на вход
	cin >> a;
        r = a / 2;
	for (int i = 0; i <= a; i++) {
		for (int j = 0; j <= a; j++) {
            if (i <= r) {
                if (j >= r - i && j <= r + i) {
                    cout << "*";
                }
                else {
                    cout << " ";
                }
            }
            else {
                if (j >= r + i - a + 1 && j <= r - i + a - 1) {
                    cout << "*";
                }
                else {
                    cout << " ";
                }
            }
		}
		cout << endl;
	}
	return 0;
}
