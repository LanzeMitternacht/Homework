#include <iostream>
using namespace std;

void distribution(int* arr, int size) {
	int* zero = new int[size];
	int* plus = new int[size];
	int* minus = new int[size];
	cout << "\n\n" << "Null position in array: " << endl;
	for (int i = 0; i < size; i++) {
		if (arr[i] == 0) {
			zero[i] = arr[i];
			cout << i << " ";
		}	
	}
	cout << "\n\n" << "Minus: " << endl;
	for (int i = 0; i < size; i++) {
		if (arr[i] < 0) {
			minus[i] = arr[i];
			cout << minus[i] << " ";
		}
	}
	cout << "\n\n" << "Plus: " << endl;
	for (int i = 0; i < size; i++) {
		if (arr[i] > 0) {
			plus[i] = arr[i];
			cout << plus[i] << " ";
		}
	}
	cout << endl;
	delete[] zero;
	delete[] plus;
	delete[] minus;
}

void random(int* arr, int size) {
	cout << "First array: " << endl;
	for (int i = 0; i < size; i++) {
		arr[i] = (rand() % 200 - 100);
		cout << arr[i] << " ";
	}
	cout << endl;
}


int main(){
	srand(time(NULL));
	int size = 100;
	int* arr = new int[size];
	random(arr, size);
	distribution(arr, size);
	delete[] arr;
	return 0;
}
