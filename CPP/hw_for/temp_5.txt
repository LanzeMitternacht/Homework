#include <iostream>
using namespace std;

int main()
{
    setlocale(LC_ALL, "rus");
    int a, b, sum = 0;
    cout << "Введите число а: ";
    cin >> a;
    if (a < 0) {
        a = -a;
    }
    b = a;
    while (a > 0) {
        sum += a % 10;
        a /= 10;
    }
    if (sum * sum * sum == b * b) {
        cout << "Да" << endl;
    }
    else {
        cout << "Нет" << endl;
    }
    return 0;
}
