#include <iostream>
using namespace std;

int main()
{
    setlocale(LC_ALL, "rus");
    int a;
    cout << "Введите число а: ";
    cin >> a;
    cout << "Числа, на которое число " << a << " делиться без остатка:\n";
    for (int i = 1; i < a; i++) {
        if (a % i == 0) {
            cout << i << endl;
        }
    }
    return 0;
}
