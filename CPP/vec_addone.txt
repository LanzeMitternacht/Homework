#include <iostream>
#include <cassert>
#include <vector>
using namespace std;

vector<int> add_one(vector<int> v) {
    for (int i = v.size()-1; i != -1; i--) {
        if (v.front() == 9) {
            v.front() = 0;
            v.insert(v.begin(), 1);
            i = v.size();
        }
        else if(v[i] == 9) {
            v[i] = 0;
            v[i - 1] += 1; 
            if (v[i - 1] >= 10) {
                v[i - 1] = 0;
            }
        }
        else if (v[i] > v.size()-1) {
            v[i] += 1;
        }
    }
    return v;
}


int main()
{
    vector<int> v{ 1,2,3 };
    vector<int> v1{9, 9, 9};
    vector<int> v2{9};
    vector<int> v3{ 1, 2, 4 };
    vector<int> v4{ 1, 0, 0, 0 };
    vector<int> v5{ 1, 0 };

    assert(add_one(v) == v3);
    assert(add_one(v1) == v4);
    assert(add_one(v2) == v5);


    cout << "Complited" << endl; 
    return 0;
}
