#include <iostream>

using namespace std;

struct Tree {
    int val;
    Tree* left;
    Tree* right;
};
int maxWeight(Tree* t) {
    if (t == NULL) {
        return 0;
    }
    return t->val + maxWeight(t->left) + maxWeight(t->right);
}


int main(){
    Tree tree{ 9, new Tree{ 8,
        new Tree{ 4, /*left*/ nullptr, /*right*/ nullptr},
        new Tree{ 1, nullptr, nullptr }
         },
        new Tree{ 3,
        new Tree{ 7,
        new Tree{2, nullptr, nullptr },
        nullptr},
        new Tree{ 5, nullptr, nullptr }
        }
    };

   int g = maxWeight(&tree);
   cout << g << endl;
    return 0;
}
