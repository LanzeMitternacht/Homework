#include <iostream>
using namespace std;

int main() {
	char flag;
	int left, right;
	int arr[101];
	for (int i = 0; i < 101; i++) {
		arr[i] = i;
		cout << arr[i] << " ";
		left = arr[0];
		right = arr[100];
	}
	cout << endl;
	while (left <= right) {
		int mid = (left + right) / 2;
		cout << "Your number more/less " << mid << "?" << endl;
		cin >> flag;
		if ((int)flag == 61) {
			cout << "Your number is " << mid;
			break;
		}
		else if ((int)flag > 60) {
			left = mid + 1;
		}
		else if ((int)flag < 62) {
			right = mid - 1;
		}
	}
	return 0;
}
