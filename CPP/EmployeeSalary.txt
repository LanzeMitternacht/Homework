#include <iostream>
#include <vector>
using namespace std;

class Employee {
    double m_salary;
public:
    Employee(double salary):m_salary(salary){}

    double getSalary() const{
        return m_salary;
    }
};

class Developer : public Employee {
public:
    Developer(double salary): Employee(salary){}
};

class Designer : public Employee {
public:
    Designer(double salary): Employee(salary){}
};

class Manager : public Employee {
public:
    Manager(double salary): Employee(salary){}
};

double sumSalary(vector<Employee*> emp) {
    double salary = 0;
    for (int i = 0; i < emp.size(); i++) {
        salary += emp[i]->getSalary();
    }
    return salary;
}
int main() {
    vector<Employee*> v{ new Developer(100), new Designer(120), new Manager(400), new Developer(3456), new Designer(234), new Manager(666), 
        new Developer(234), new Designer(123), new Manager(890), new Developer(5678), new Designer(34), new Manager(35)};
    
    double salary = sumSalary(v);
    cout << "Salary about v: " << salary << "$" << endl;
    
    return 0;
}
