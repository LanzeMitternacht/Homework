#include <iostream>
#include <vector>
#include <set>
#include <algorithm>
using namespace std;


int main() {
	vector<int> vec = { 1,2,3,4,5,5,6,7,8 };
	vector<int> vec1 = { 1,2,3,4,4,4,5,6,7,8 };
	vector<int> vec2 = { 5, 6, 7, 8, 9 };
	vector<int> resultVec;
	set<int> vecNumbers;

	for_each(vec.begin(), vec.end(), [&vecNumbers](int num) {
		vecNumbers.insert(num);
	});
	for_each(vec1.begin(), vec1.end(), [&vecNumbers](int num) {
		vecNumbers.insert(num);
	});
	for_each(vec2.begin(), vec2.end(), [&vecNumbers](int num) {
		vecNumbers.insert(num);
	});
	
	copy(vecNumbers.begin(), vecNumbers.end(), back_inserter(resultVec));
	

	cout << "Vector with non-repeating numbers: ";
	for (int num : resultVec) {
		cout << num << " ";
	}
	cout << endl;
	return 0;
}
